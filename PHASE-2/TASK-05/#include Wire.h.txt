#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <ESP32Servo.h>
#include <DS1302.h>

// ===== LCD =====
LiquidCrystal_I2C lcd(0x3F, 16, 2);

// ===== RTC =====
#define CLK 5
#define DAT 4
#define RST 2
DS1302 rtc(RST, DAT, CLK);

// ===== SERVOS =====
Servo servo1;
Servo servo2;

#define SERVO1_PIN 18
#define SERVO2_PIN 23

// ===== BUZZER =====
#define BUZZER 19

// ===== KEYPAD =====
int rowPins[4] = {13,12,14,27};
int colPins[4] = {26,25,33,32};

char keys[4][4] = {
{'1','2','3','A'},
{'4','5','6','B'},
{'7','8','9','C'},
{'*','0','#','D'}
};

// ===== SCHEDULE STORAGE =====
String slotTime[4];
int slotServo[4];
bool slotDone[4];

// ===== GLOBAL =====
bool alarmActive = false;
int activeSlot = -1;

// =================================================
void setup() {

Serial.begin(115200);
Wire.begin(21,22);

lcd.init();
lcd.backlight();

servo1.attach(SERVO1_PIN);
servo2.attach(SERVO2_PIN);

servo1.write(90);
servo2.write(90);

pinMode(BUZZER,OUTPUT);

for(int i=0;i<4;i++){
pinMode(rowPins[i],OUTPUT);
digitalWrite(rowPins[i],HIGH);
pinMode(colPins[i],INPUT_PULLUP);
}

rtc.halt(false);
rtc.writeProtect(false);

// ==== SET TIME ONLY ONCE ====
//Time t(2026,3,6,19,20,0,Time::kThursday);
//rtc.time(t);

lcd.print("Medicine System");
delay(2000);
lcd.clear();

startupBeep();
}

// =================================================
void loop() {

showTime();

char key = scanKeypad();

if(key=='D'){
programSlot();
}

checkAlarm();

if(alarmActive){
handleAlarm();
}

delay(200);
}

// =================================================
void startupBeep(){
for(int i=0;i<4;i++){
digitalWrite(BUZZER,HIGH);
delay(200);
digitalWrite(BUZZER,LOW);
delay(200);
}
}

// =================================================
char scanKeypad(){

for(int r=0;r<4;r++){
digitalWrite(rowPins[r],LOW);

for(int c=0;c<4;c++){

if(digitalRead(colPins[c])==LOW){
delay(200);
digitalWrite(rowPins[r],HIGH);
return keys[r][c];
}
}

digitalWrite(rowPins[r],HIGH);
}

return 0;
}

// =================================================
void programSlot(){

lcd.clear();
lcd.print("Slot 1-4 ?");

int slot=-1;

while(slot==-1){

char k=scanKeypad();

if(k>='1' && k<='4'){
slot=k-'1';
}
}

lcd.clear();
lcd.print("Servo 1 or 2");

int servo=-1;

while(servo==-1){

char k=scanKeypad();

if(k=='1') servo=1;
if(k=='2') servo=2;
}

slotServo[slot]=servo;

lcd.clear();
lcd.print("Time HHMM");

String t="";

while(t.length()<4){

char k=scanKeypad();

if(isDigit(k)){
t+=k;
lcd.setCursor(0,1);
lcd.print(t);
}
}

slotTime[slot]=t;
slotDone[slot]=false;

lcd.clear();
lcd.print("Saved Slot ");
lcd.print(slot+1);

delay(2000);
lcd.clear();
}

// =================================================
void checkAlarm(){

Time t = rtc.time();

String hh=(t.hr<10?"0":"")+String(t.hr);
String mm=(t.min<10?"0":"")+String(t.min);

String now=hh+mm;

for(int i=0;i<4;i++){

if(slotTime[i]==now && !slotDone[i]){

alarmActive=true;
activeSlot=i;
slotDone[i]=true;
}
}
}

// =================================================
void handleAlarm(){

digitalWrite(BUZZER,HIGH);

lcd.setCursor(0,0);
lcd.print("Medicine Time ");

lcd.setCursor(0,1);
lcd.print("Press *");

char key=scanKeypad();

if(key=='*'){

digitalWrite(BUZZER,LOW);

dispenseMedicine(activeSlot);

alarmActive=false;
}

}

// =================================================
void dispenseMedicine(int s){

lcd.clear();
lcd.print("Dispensing");

if(slotServo[s]==1){

servo1.write(0);
delay(1500);
servo1.write(90);

}

if(slotServo[s]==2){

servo2.write(180);
delay(1500);
servo2.write(90);

}

delay(2000);

lcd.clear();
lcd.print("Done");

delay(2000);
lcd.clear();
}

// =================================================
void showTime(){

Time t = rtc.time();

int hr=t.hr;
String ampm="AM";

if(hr>=12){
ampm="PM";
}

if(hr>12){
hr=hr-12;
}

if(hr==0){
hr=12;
}

lcd.setCursor(0,0);

lcd.print("Time ");

if(hr<10) lcd.print("0");
lcd.print(hr);
lcd.print(":");

if(t.min<10) lcd.print("0");
lcd.print(t.min);
lcd.print(":");

if(t.sec<10) lcd.print("0");
lcd.print(t.sec);

lcd.print(" ");
lcd.print(ampm);

lcd.print("   ");

lcd.setCursor(0,1);

lcd.print("Slots Ready   ");
}
